% \textipa{}, {\tipaencoding} e ambiente IPA
\begin{frame}[fragile]
  \frametitle{Linguística: \latexcode{tipa}}
  \huge

  Três modos de acessar:
  \begin{onlyenv}<1>
    \begin{minted}[autogobble,breaklines]{latex}
      \textipa{…}
    \end{minted}
  \end{onlyenv}

  \begin{onlyenv}<2>
    \begin{minted}[autogobble,breaklines]{latex}
      {\tipaencoding …}
    \end{minted}
  \end{onlyenv}

  \begin{onlyenv}<3>
    o ambiente \latexcode{IPA}
  \end{onlyenv}
\end{frame}

% Exemplos básicos
\begin{frame}[fragile]
  \frametitle{Linguística: exemplo com o \latexcode{tipa}}
  \huge

  \begin{minted}[autogobble,fontsize=\large,breaklines]{latex}
    \textipa{l\~\i N."g\super wis.t5s}
  \end{minted}

  \textipa{l\~\i N."g\super wis.t5s}%
  \footnote{Exemplos retirados de
  \url{https://social.stoa.usp.br/latex-para-linguistas/}}

\end{frame}

% Vogais orais do ptbr
\begin{frame}[fragile]

  \huge

  \begin{minted}[autogobble,fontsize=\Large,breaklines]{latex}

  \end{minted}

  \textipa{[I U 5]}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Linguística: exemplos com o \latexcode{linguex}}

  \end{columns}
\end{frame}

% Glosas (de https://www.eva.mpg.de/lingua/resources/glossing-rules.php)
\begin{frame}[fragile]
  \frametitle{Linguística: glosas com o \latexcode{linguex}}
  \small

  \begin{minted}[autogobble,breaklines]{latex}
    \exg.
    Gila abur-u-n ferma hamišaluǧ güǧüna amuq’-da-č.\\
    now they-OBL-GEN farm forever behind stay-FUT-NEG\\
    ``Now their farm will not stay behind forever.''
  \end{minted}

  \exg.
  Gila abur-u-n ferma hamišaluǧ güǧüna amuq’-da-č.\\

  ``Now their farm will not stay behind forever.''%

\begin{frame}[fragile]

  \normalsize

  \begin{minted}[autogobble,fontsize=\small,breaklines]{latex}
    \ex.\label{ex:gram} Chegou o menino.

    \ex.*\label{ex:agram} A chegou carta.

    Em~\ref{ex:gram}, temos uma sentença bem formada, ao contrário de~\ref{ex:agram}, que é uma sentença agramatical.
  \end{minted}
